sizes = top10.count .* 10  # Tailles des bulles

labels = coalesce.(top10.track_name, "")

    title = "Top musiques",

    ylabel = ""
)

    ax,
    positions;
    markersize = sizes,
    color = colors,
    colormap = :viridis,
    strokewidth = 0.8,
    strokecolor = :black
)
text!(
    ax,
    positions;
    text = labels,
    font = "bold",
