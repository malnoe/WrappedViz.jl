top10_artists_time =
    combine(groupby(df, :artist),
            :ms_played => sum => :total_time) |>
    df -> transform(df, :total_time => ByRow(x -> x / 60000) => :total_time_min) |>
    df -> sort(df, :total_time_min, rev = true) |>
    df -> df[1:10, :]
